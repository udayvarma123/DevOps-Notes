Jenkins installation

curl --silent --location http://pkg.jenkins-ci.org/redhat-stable/jenkins.repo | sudo tee /etc/yum.repos.d/jenkins.repo
sudo rpm --import https://jenkins-ci.org/redhat/jenkins-ci.org.key
yum install jenkins --> install the latest version
OR
yum install jenkins-2.401.3-1.1 --> install a specific version
systemctl start jenkins 
systemctl status jenkins
systemctl enable jenkins
cat /var/lib/jenkins/secrets/initialAdminPassword

sudo chmod +x /usr/local/bin/docker-compose

k8 installation - 

Perform on master --

hostnamectl set-hostname master
setenforce 0
sed -i --follow-symlinks 's/SELINUX=enforcing/SELINUX=disabled/g' /etc/sysconfig/selinux
swapoff -a 

cat <<EOF > /etc/yum.repos.d/kubernetes.repo
[kubernetes]
name=Kubernetes
baseurl=https://packages.cloud.google.com/yum/repos/kubernetes-el7-x86_64
enabled=1
gpgcheck=1
repo_gpgcheck=1
gpgkey=https://packages.cloud.google.com/yum/doc/yum-key.gpg https://packages.cloud.google.com/yum/doc/rpm-package-key.gpg
EOF

yum -v list kubelet --show-duplicates - optional
yum install -y docker
yum install kubelet-1.20.12-0 kubeadm-1.20.12-0 kubectl-1.20.12-0 -y
kubeadm init --pod-network-cidr=10.244.0.0/16
mkdir -p $HOME/.kube
sudo cp -i /etc/kubernetes/admin.conf $HOME/.kube/config
sudo chown $(id -u):$(id -g) $HOME/.kube/config

kubectl apply -f https://raw.githubusercontent.com/coreos/flannel/master/Documentation/kube-flannel.yml

systemctl start kubelet
systemctl start docker

systemctl enable kubelet
systemctl enable docker

kubectl get nodes

Perform on agent - 

hostnamectl set-hostname agent
setenforce 0
sed -i --follow-symlinks 's/SELINUX=enforcing/SELINUX=disabled/g' /etc/sysconfig/selinux
swapoff -a 

cat <<EOF > /etc/yum.repos.d/kubernetes.repo
[kubernetes]
name=Kubernetes
baseurl=https://packages.cloud.google.com/yum/repos/kubernetes-el7-x86_64
enabled=1
gpgcheck=1
repo_gpgcheck=1
gpgkey=https://packages.cloud.google.com/yum/doc/yum-key.gpg https://packages.cloud.google.com/yum/doc/rpm-package-key.gpg
EOF

yum -v list kubelet --show-duplicates - optional step
yum install docker -y
yum install kubectl-1.20.12-0 kubelet-1.20.12-0 kubeadm-1.20.12-0 -y

systemctl start kubelet
systemctl start docker

systemctl enable kubelet
systemctl enable docker

Terraform installation

sudo yum install -y yum-utils
sudo yum-config-manager --add-repo https://rpm.releases.hashicorp.com/RHEL/hashicorp.repo
sudo yum -y install terraform

				OR

./terraform
cp terraform /usr/local/bin/
terraform --version
